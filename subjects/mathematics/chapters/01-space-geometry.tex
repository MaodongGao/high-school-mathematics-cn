\section{点、直线、平面}

平面是向四周无限延展的；图中常画成平行四边形，记作平面 $\alpha$，也可用其中不共线的三点记作平面 $ABC$。基本语言如下。

\begin{center}
  \begin{tabular}{llll}
    \toprule
    关系 & 记号 & 关系 & 记号 \\
    \midrule
    点 $A$ 在直线 $l$ 上 & $A\in l$ & 点 $A$ 不在 $l$ 上 & $A\notin l$ \\
    点 $A$ 在平面 $\alpha$ 内 & $A\in\alpha$ & 点 $A$ 不在 $\alpha$ 内 & $A\notin\alpha$ \\
    直线 $l$ 在平面 $\alpha$ 内 & $l\subset\alpha$ & 直线 $l$ 与 $\alpha$ 相交 & $l\cap\alpha=A$ \\
    两平面交于直线 $l$ & $\alpha\cap\beta=l$ & 两平面平行 & $\alpha\parallel\beta$ \\
    \bottomrule
  \end{tabular}
\end{center}

点与直线之间用 $\in$；直线与平面之间用 $\subset$。不要写成 $l\in\alpha$。

\subsection{平面的三个基本性质}

\begin{lawbox}[点线面的基本性质]
  \begin{enumerate}
    \item 若直线 $l$ 上两点 $A,B$ 在平面 $\alpha$ 内，则 $l\subset\alpha$。
    \item 不在同一直线上的三点确定唯一平面。
    \item 若两个不重合平面有一个公共点 $P$，则它们相交于一条过 $P$ 的直线：$\alpha\cap\beta=l$，$P\in l$。
  \end{enumerate}
\end{lawbox}

\begin{corollarybox}[确定平面的三种方式]
  一条直线和线外一点、两条相交直线、两条平行直线，都确定唯一平面。
  \begin{center}
  \begin{tikzpicture}[scale=.58]
    \draw[sg edge] (0,0)--(2.5,0); \node[sg mark,label={[sg label]above:$P$}] at (1.25,1) {};
    \node[font=\small] at (1.25,-.5) {线 $+$ 线外点};
    \draw[sg edge] (3.7,0)--(6.2,1.25) (4.15,1.2)--(6.1,-.15);
    \node[font=\small] at (5,-.5) {两相交线};
    \draw[sg edge] (7.6,.15)--(10.1,.15) (7.9,1.15)--(10.4,1.15);
    \node[font=\small] at (9,-.5) {两平行线};
  \end{tikzpicture}
  \end{center}
\end{corollarybox}

\begin{corollarybox}[点线面证明入口]
  证线在面内：在线上找两点；证若干点共面：先确定一个平面，再逐点验证；证空间点共线：把各点同时放入两个相交平面，公共点必在交线上。
\end{corollarybox}

\begin{examplebox}[原稿例：交线法证明共线]
  在四面体 $ABCD$ 中，两条分别位于面 $ABD$、面 $BCD$ 内的直线交于点 $P$。若已知 $P$ 同时在这两个面内，证明 $P,B,D$ 共线。

  \begin{center}
  \begin{tikzpicture}[scale=.62]
    \coordinate (B) at (0,0); \coordinate (D) at (4.8,.6); \coordinate (A) at (1.8,2.4); \coordinate (C) at (2.7,-1.55); \coordinate (P) at (3,.375);
    \path[draw=BookBlue,fill=BookBlueLight] (A)--(B)--(D)--cycle;
    \path[draw=BookGold,fill=BookGold!10,fill opacity=.75] (C)--(B)--(D)--cycle;
    \draw[very thick,BookGold] (B)--(D);
    \node[sg mark] at (P) {};
    \foreach \q/\pos in {A/above,B/left,C/below,D/right,P/above}\node[sg label,\pos] at (\q) {$\q$};
    \node[font=\small,align=left] at (7,.5) {$\text{面 }ABD\cap\text{面 }BCD=BD$\\$P$ 同属两面 $\Rightarrow P\in BD$};
  \end{tikzpicture}
  \end{center}

  \textbf{解：}$P\in\text{平面 }ABD$ 且 $P\in\text{平面 }BCD$；而两平面的交线为 $BD$，故 $P\in BD$。因此 $P,B,D$ 共线。

  \textbf{方法启示：}证明空间点共线时，可以先寻找两个相交平面，再对目标点逐一建立“双重归属”；点同时属于两平面，也就属于它们的交线。
\end{examplebox}

\section{空间两直线的位置关系}

两直线若在同一平面内，则相交或平行；若不同在任何一个平面内，则称为\textbf{异面直线}。因此空间两直线只有三种位置关系：
\[
  \text{相交}\quad/\quad\text{平行}\quad/\quad\text{异面}.
\]

\begin{lawbox}[异面直线的判定]
  若直线 $a$ 与平面 $\alpha$ 相交于 $A$，直线 $b\subset\alpha$ 且 $A\notin b$，则 $a,b$ 异面。
\end{lawbox}

证明只需反证：若 $a,b$ 共面于 $\beta$，则 $A\in a\subset\beta$ 且 $b\subset\alpha\cap\beta$；由于 $A\notin b$，两平面的交线不能同时容纳 $A$ 与 $b$，矛盾。

\begin{corollarybox}[空间平行线的传递性]
  若 $a\parallel b$、$b\parallel c$，则 $a\parallel c$。三条线可以不在同一平面内；只要都与同一方向平行，任取其中两条都能确定一个平面并在该平面内平行。
\end{corollarybox}

\subsection{异面直线所成的角与公垂线}

过空间任一点 $O$ 分别作 $a'\parallel a$、$b'\parallel b$，相交直线 $a',b'$ 所成的锐角或直角称为异面直线 $a,b$ 所成的角。

\begin{corollarybox}[空间等角定理]
  若两个角的两边分别平行且方向相同，则两角相等；若一组方向相反，则两角互补。因此平移所选点不会改变异面直线所成的锐角或直角。
\end{corollarybox}

若所成角为 $90^\circ$，记作 $a\perp b$；这一定义同时适用于相交直线和异面直线。两条异面直线有且只有一条公垂线段，其长度称为两异面直线间的距离。

\begin{examplebox}[原稿型：正方体中的异面线夹角]
  正方体 $ABCD-A_1B_1C_1D_1$ 的棱长为 $a$，求异面直线 $A_1B$ 与 $B_1C$ 所成的角。

\section{平行关系}

\subsection{直线与平面平行}

若直线 $a$ 与平面 $\alpha$ 没有公共点，称 $a\parallel\alpha$。画图时把 $a$ 画成与平面内某条直线平行。

\begin{lawbox}[线面平行的判定与性质]
  \textbf{判定：}若 $a\not\subset\alpha$，$b\subset\alpha$ 且 $a\parallel b$，则 $a\parallel\alpha$。

  \textbf{性质：}若 $a\parallel\alpha$，过 $a$ 的平面 $\beta$ 与 $\alpha$ 交于 $b$，则 $a\parallel b$。
\end{lawbox}

\textbf{为什么只需找一条平行线？} 过 $a,b$ 作平面 $\beta$。若 $a$ 还与 $\alpha$ 相交，交点必须同时落在
$\alpha\cap\beta=b$ 上，这与 $a\parallel b$ 矛盾。因此 $a$ 不碰 $\alpha$。反过来，用过 $a$ 的平面去截
$\alpha$，截线 $b$ 与 $a$ 共面且不能相交，所以只能平行。

\begin{corollarybox}[线面平行与线线平行互化]
  判定时在平面内“找平行线”；性质则用过已知直线的辅助平面“截出平行线”：
  \[
    \text{线面平行}\ \Longleftrightarrow\ \text{线线平行（配合辅助平面）}.
  \]
\end{corollarybox}

\subsection{平面与平面平行}

若 $\alpha,\beta$ 没有公共点，称 $\alpha\parallel\beta$。

\begin{lawbox}[面面平行的判定]
  若平面 $\alpha$ 内两条相交直线 $a,b$ 都平行于平面 $\beta$，则 $\alpha\parallel\beta$。

  等价地，若 $a,b\subset\alpha$，$a',b'\subset\beta$，$a\cap b=O$，$a'\cap b'=O'$，且 $a\parallel a'$、$b\parallel b'$，则 $\alpha\parallel\beta$。
\end{lawbox}

这里必须是\textbf{两条相交直线}：一条线只锁定一个方向，平面仍可绕它“转动”；两个相交方向才能锁定整个平面的方向。

\begin{lawbox}[面面平行的性质]
  \begin{enumerate}
    \item 若 $\alpha\parallel\beta$，则 $\alpha$ 内任一直线都平行于 $\beta$。
    \item 若 $\alpha\parallel\beta$，平面 $\gamma$ 分别与它们交于 $a,b$，则 $a\parallel b$。
    \item 夹在两个平行平面间的平行线段相等；两平行平面间的距离处处相等。
  \end{enumerate}
\end{lawbox}

\section{垂直关系}

\subsection{直线与平面垂直}

若直线 $l$ 与平面 $\alpha$ 内任意直线都垂直，称 $l\perp\alpha$；交点称为垂足。

\begin{lawbox}[线面垂直的判定]
  若 $m,n\subset\alpha$，$m\cap n=O$，且 $l\perp m$、$l\perp n$，则 $l\perp\alpha$。
\end{lawbox}

\textbf{理解：}平面内两条相交直线给出两个独立方向。$l$ 同时垂直这两个方向，就垂直于平面内由它们确定的任意方向，因此垂直于整个平面。

“两条相交直线”不可省略：只垂直于平面内一条直线，不能推出垂直于该平面。

\begin{corollarybox}[垂直关系的平行传递]
  \[
    l\perp\alpha,\ m\parallel l\Rightarrow m\perp\alpha;\qquad
    l\perp\alpha,\ m\perp\alpha\Rightarrow l\parallel m;
  \]
  \[
    l\perp\alpha,\ l\perp\beta\Rightarrow\alpha\parallel\beta.
  \]

    \path[sg plane] (0,0)--(3.4,0)--(4.1,1)--(.7,1)--cycle;

  \end{tikzpicture}
  \end{center}
\end{corollarybox}

\begin{corollarybox}[垂线与垂面的唯一性]
  过空间一点有且只有一条直线垂直于已知平面；过空间一点有且只有一个平面垂直于已知直线。
\end{corollarybox}

\subsection{斜线、射影与三垂线定理}

设 $P\notin\alpha$，$PO\perp\alpha$。$PO$ 是垂线段，$O$ 是 $P$ 在 $\alpha$ 上的射影；对 $A\in\alpha$、$A\ne O$，$PA$ 是斜线段，$OA$ 是 $PA$ 在 $\alpha$ 内的射影。

\begin{corollarybox}[斜线段与射影的大小关系]
  从同一点 $P$ 向平面 $\alpha$ 引两条斜线 $PA,PB$，因
  \[
    PA^2=PO^2+OA^2,\qquad PB^2=PO^2+OB^2,
  \]
  所以 $PA=PB\Longleftrightarrow OA=OB$，且 $PA>PB\Longleftrightarrow OA>OB$。比较斜线长短，可转化为比较平面内射影。
\end{corollarybox}

\begin{lawbox}[三垂线定理及逆定理]
  对平面内过 $A$ 的直线 $a$：
  \[
    a\perp OA\ \Longrightarrow\ a\perp PA,
    \qquad
    a\perp PA\ \Longrightarrow\ a\perp OA.
  \]
  即：平面内一条直线垂直于斜线的射影，当且仅当它垂直于斜线本身。
\end{lawbox}

\textbf{证明骨架：}因 $PO\perp\alpha$，故 $a\perp PO$。若再有 $a\perp OA$，则 $a$ 垂直于平面
$POA$ 内两条相交直线 $PO,OA$，从而 $a\perp$ 平面 $POA$，特别地 $a\perp PA$。逆定理同理把
$PA$ 与 $PO$ 作为两条相交直线。

\subsection{二面角与面面垂直}

\begin{lawbox}[面面垂直的判定与性质]
  \textbf{判定：}若平面 $\beta$ 经过平面 $\alpha$ 的一条垂线，则 $\alpha\perp\beta$。

  \textbf{性质：}若 $\alpha\perp\beta$，$\alpha\cap\beta=l$，直线 $a\subset\alpha$ 且 $a\perp l$，则 $a\perp\beta$。
\end{lawbox}

\subsection{空间直线的方向余弦}

设直线的一个方向向量为 $\vec v=(x,y,z)$。它在三个坐标轴上的投影长度分别为 $x,y,z$，所以直线与
$x,y,z$ 轴所成角 $\alpha,\beta,\gamma$ 满足
\[
  \cos\alpha=\frac{x}{\sqrt{x^2+y^2+z^2}},\qquad
  \cos\beta =\frac{y}{\sqrt{x^2+y^2+z^2}},\qquad
  \cos\gamma=\frac{z}{\sqrt{x^2+y^2+z^2}}.
\]
\begin{corollarybox}[方向角的平方和]
  方向余弦就是把方向向量化为单位向量后的三个坐标。由单位向量长度为 $1$，
  \[
    \cos^2\alpha+\cos^2\beta+\cos^2\gamma=1,
    \qquad
    \sin^2\alpha+\sin^2\beta+\sin^2\gamma=2.
  \]
  第二式由 $\sin^2\theta=1-\cos^2\theta$ 对三个角相加得到，不必另记。
\end{corollarybox}

\begin{examplebox}[方向向量读出方向余弦]
  若直线方向向量为 $(1,2,2)$，其长度为 $3$，故
  \begin{center}
  \begin{tikzpicture}[scale=.62]
    \coordinate (O) at (0,0); \coordinate (X) at (3.4,0); \coordinate (Y) at (-1.2,-.8); \coordinate (Z) at (0,3.1); \coordinate (P) at (2.35,2.25);

    \draw[sg edge,-{Latex[length=2mm]}] (O)--(Y) node[left,sg label] {$y$};
    \draw[sg edge,-{Latex[length=2mm]}] (O)--(Z) node[above,sg label] {$z$};
    \draw[very thick,BookGold,-{Latex[length=2mm]}] (O)--(P) node[above right,sg label] {$\vec v$};

  \[

\section{多面体}

由若干平面多边形围成的几何体称为多面体；围成它的多边形是面，相邻面的公共边是棱，棱的公共端点是顶点。若多面体位于每个面的同一侧，称为凸多面体，否则为凹多面体。

\subsection{棱柱、棱锥与棱台}

\begin{description}
  \item[棱柱] 两个面互相平行，其余各面为平行四边形，且相邻侧面的公共边互相平行。平行的两个面是底面，其余为侧面。侧棱垂直底面的棱柱是直棱柱；底面为正多边形的直棱柱是正棱柱。
  \item[平行六面体] 底面为平行四边形的四棱柱。六个面都是矩形时为长方体；六个面都是正方形时为正方体。
  \item[棱锥] 一个面是多边形，其余各面是有公共顶点的三角形。若底面为正多边形且顶点在底面的射影为底面中心，称为正棱锥。
  \item[棱台] 用平行于棱锥底面的平面截棱锥，截面与原底面之间的部分称为棱台；由正棱锥截得的是正棱台。
\end{description}

  正棱锥的侧棱相等，各侧面是全等的等腰三角形，各侧面与底面所成的二面角也相等；侧面底边上的高称为斜高 $l$。设底面周长为 $p$，则

\subsection{祖氏原理与统一体积公式}

\begin{lawbox}[祖氏原理（等幂等积）]
  夹在两个平行平面之间的两个几何体，若在任意等高处的截面面积都相等，则两几何体体积相等。
\end{lawbox}

\begin{corollarybox}[柱、锥、台的统一体积公式]
  \[
    V_{\text{柱体}}=Sh,
    \qquad
    V_{\text{锥体}}=\frac13Sh,
    \qquad
    V_{\text{台体}}=\frac h3\left(S+\sqrt{SS'}+S'\right).
  \]
  $S,S'$ 是上、下底面积；台体退化为柱体时 $S=S'$，退化为锥体时 $S'=0$，公式会自动回到前两式，可用于验算。
\end{corollarybox}

三条公式的来源不同但脉络相同：柱体各等高截面都与底面等积，所以体积是 $Sh$；一个三棱柱可以分成三个等体积三棱锥，这给出锥体前的系数 $1/3$，再由祖氏原理推广到同底等高的锥体；台体则补成大锥体，再减去顶部小锥体。先辨认“柱、锥、台”，再使用对应系数。

台体公式来自补成大锥体。若小锥与大锥相似，底面积分别为 $S',S$，对应高为 $x,x+h$，则
\[
  \frac{x}{x+h}=\sqrt{\frac{S'}S};
\]
用“大锥体积减小锥体积”消去 $x$ 即得台体公式。

\begin{corollarybox}[体积题的三条路线]
  直接套公式；换底或祖氏原理作等体积转化；把复杂几何体分割、补形为熟悉几何体。遇到空间高难求时，优先尝试“换底”。
\end{corollarybox}

\begin{examplebox}[原稿方法：四面体换底]
  四面体体积既可把 $\triangle ABC$ 作为底，也可把 $\triangle PAB$ 作为底：

\section{球}

半圆绕直径旋转所得的几何体是球；半圆弧旋转所得曲面是球面。球心记为 $O$，半径为 $R$，直径为 $2R$。

球面被任一平面截得的截线是圆。

\begin{corollarybox}[球的核心截面]
  若球心到截面的距离为 $d$、截面圆半径为 $r$，连接球心、截面圆心和截面圆上一点，得到直角三角形，因此
  \[
    \boxed{r^2+d^2=R^2}.
  \]
  球的截面、内切与外接问题，通常都先压缩到这个平面图。
\end{corollarybox}

  \[
    S_{\text{球}}=4\pi R^2,

\end{corollarybox}

\subsection{内切球与外接球}

\begin{corollarybox}[多面体内切球的体积分割]
  球面与多面体各面相切时，球为内切球。以球心为顶点、各面为底分割多面体，每个小棱锥的高都等于内切球半径 $r$，所以
  \[
    \boxed{V=\frac13rS_{\text{表}}}.
  \]
\end{corollarybox}

\begin{corollarybox}[正四面体的内外接球半径]
  正四面体棱长为 $a$ 时，内、外球心重合并位于高线上，且
  \[
    h=\frac{\sqrt6}{3}a,
    \quad r=\frac h4=\frac{\sqrt6}{12}a,
    \quad R=\frac{3h}{4}=\frac{\sqrt6}{4}a,
    \quad \boxed{R=3r}.
  \]
\end{corollarybox}

处理内外接球题，先找对称轴与截面：把空间问题压缩为包含球心、切点或顶点的平面图形，再用勾股定理。

\section{投影、三视图与直观图}

光线照射物体，在平面上形成的影子是投影。投影线交于一点的是中心投影；投影线互相平行的是平行投影。平行投影又分为斜投影和正投影（投影线垂直投影面）。

\begin{corollarybox}[平行投影的保持量]
  平行投影保持共线性和平行性；与投影面平行的线段、平面图形保持真实长度与形状。它一般不保持角度，也不保持不与投影面平行的长度。
\end{corollarybox}

\subsection{三视图}

正视图、侧视图、俯视图分别反映物体的
\[
  \text{长与高},\qquad\text{宽与高},\qquad\text{长与宽}.

\begin{corollarybox}[三视图的尺寸对应]

\section{三余弦定理}

设 $P\notin\alpha$，$PO\perp\alpha$，$A,B\in\alpha$。$OA$ 是斜线 $PA$ 在平面 $\alpha$ 内的射影。令
\[
  \angle PAO=\theta,
  \qquad
  \angle OAB=\varphi,
  \qquad
  \angle PAB=\psi.
\]

\begin{lawbox}[三余弦定理]
  \[
    \boxed{\cos\psi=\cos\theta\cos\varphi}.
  \]
  即：斜线与平面内直线夹角的余弦，等于斜线与其射影夹角的余弦乘以射影与该直线夹角的余弦。
\end{lawbox}

\begin{center}
\begin{tikzpicture}[scale=.92]
  \path[sg plane] (0,0)--(6.1,0)--(7.1,1.75)--(1,1.75)--cycle;
  \coordinate (O) at (2.35,.8); \coordinate (A) at (4.6,.8); \coordinate (B) at (5.75,1.45); \coordinate (P) at (2.35,3.6);
  \draw[sg edge] (P)--(O)--(A)--(P) (A)--(B);
  \draw[sg hidden] (O)--(B);
  \draw[sg aux] (O)++(.17,0)--++(0,.17)--++(-.17,0);
  \pic[draw=BookGold,angle radius=6mm,"$\theta$" {font=\scriptsize}] {angle=P--A--O};
  \pic[draw=BookBlue,angle radius=8mm,"$\varphi$" {font=\scriptsize}] {angle=O--A--B};
  \pic[draw=BookGold,angle radius=11mm,"$\psi$" {font=\scriptsize}] {angle=P--A--B};
  \foreach \p/\pos in {P/above,O/below,A/below,B/right}\node[sg label,\pos] at (\p) {$\p$};
  \node[font=\small,align=left] at (9.2,1.75) {$PA$ 的射影是 $OA$\\[3pt]
    $\theta=\angle PAO$\\$\varphi=\angle OAB$\\$\psi=\angle PAB$\\[4pt]
    $\boxed{\cos\psi=\cos\theta\cos\varphi}$};
\end{tikzpicture}
\sgcaption{三余弦定理同时追踪“斜线—射影—面内目标线”三个角。}
\end{center}

证明只用正射影。因为 $PO\perp AB$，
\[
  \overrightarrow{AP}\cdot\overrightarrow{AB}
  =\left(\overrightarrow{AO}+\overrightarrow{OP}\right)\cdot\overrightarrow{AB}
  =\overrightarrow{AO}\cdot\overrightarrow{AB}.
\]
两边除以 $|AP|\,|AB|$，再插入 $|AO|/|AP|=\cos\theta$，即得结论。

\begin{corollarybox}[三余弦定理包含三垂线定理]
  令 $\psi=90^\circ$，则 $\cos\psi=0$。因斜线与射影夹角 $\theta<90^\circ$，$\cos\theta\ne0$，所以
  $\cos\varphi=0$，即 $OA\perp AB$；反向推理同样成立。这正是三垂线定理及其逆定理。
\end{corollarybox}
